% Custom functions
\tikzmath{
	function sinc(\x) {
		if  abs(\x) < .001 then { % (|x| < .001) ~ (x = 0)
			return 1;
		} else {
			return sin(\x r)/\x;
		};
	};
}
\tikzmath{
	function asinc(\x, \y) {
		if  abs(\x) < .001 then { % (|x| < .001) ~ (x = 0)
			return 1;
		} else {
			return sin(\y * deg(\x) / 2)/(\y * sin(deg(\x)/2));
		};
	};
}
